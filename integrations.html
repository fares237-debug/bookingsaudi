// إنشاء عقد إيجار تلقائياً في منصة إيجار async function createEjarContract(booking) { const contract = { propertyId: booking.propertyEjarId, // رقم العقار في إيجار tenantId: booking.guestNationalId, // هوية المستأجر ownerId: booking.ownerNationalId, startDate: booking.checkIn, endDate: booking.checkOut, monthlyRent: booking.monthlyPrice, // السعر الشهري contractType: 'RESIDENTIAL_HOTEL', // نوع العقد paymentMethod: 'MADAD' // دفع مدد }; const res = await fetch('https://api.ejar.sa/v2/contracts', { method: 'POST', headers: { 'Authorization': `Bearer ${EJAR_TOKEN}` }, body: JSON.stringify(contract) }); return res.json(); // { contractId, status: 'PENDING_SIGNATURE' } }
